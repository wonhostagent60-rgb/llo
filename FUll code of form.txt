<script>
function goTo(id) {
  document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
  document.getElementById(id).classList.add('active');
  window.scrollTo(0, 0);
}

function submitForm(color) {
  const p = color === 'blue' ? 'b' : 'g';
  let ok = true;
  const fields = [
    { id: p+'-user',  err: p+'-e-user',  check: v => v.trim() !== '' },
    { id: p+'-age',   err: p+'-e-age',   check: v => v !== '' && +v >= 1 && +v <= 120 },
    { id: p+'-email', err: p+'-e-email', check: v => /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(v) }
  ];
  fields.forEach(f => {
    const inp = document.getElementById(f.id);
    const em  = document.getElementById(f.err);
    if (!f.check(inp.value)) { inp.classList.add('err'); em.classList.add('show'); ok = false; }
    else { inp.classList.remove('err'); em.classList.remove('show'); }
  });
  if (!ok) return;
  const btn = document.getElementById(p + '-btn');
  btn.disabled = true;
  btn.textContent = 'Verifying...';
  setTimeout(() => goTo('page-' + color + '-success'), 1800);
}
</script>
