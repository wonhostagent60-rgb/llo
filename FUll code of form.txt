<script>
function submitForm() {
  let ok = true;
  const fields = [
    { id: 'username', err: 'e-username', check: v => v.trim() !== '' },
    { id: 'age',      err: 'e-age',      check: v => v !== '' && +v >= 1 && +v <= 120 },
    { id: 'email',    err: 'e-email',    check: v => /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(v) }
  ];
  fields.forEach(f => {
    const inp = document.getElementById(f.id);
    const em = document.getElementById(f.err);
    if (!f.check(inp.value)) { inp.classList.add('err'); em.classList.add('show'); ok = false; }
    else { inp.classList.remove('err'); em.classList.remove('show'); }
  });
  if (!ok) return;
  const btn = document.getElementById('continueBtn');
  btn.disabled = true; btn.textContent = 'Verifying...';
  setTimeout(() => {
    document.getElementById('formPage').style.display = 'none';
    document.getElementById('successPage').style.display = 'block';
    window.scrollTo(0, 0);
  }, 1800);
}
</script>
